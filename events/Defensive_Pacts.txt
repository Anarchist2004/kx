add_namespace = defensive_pacts

#SA AGAINST PANAMA
country_event = { #pact against PAN
	id = defensivepact.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event sagen.1" }
	title = defensivepact.1.t
	desc = defensivepact.1.d

	fire_only_once = yes

	trigger = {	
		country_exists = PAN
		is_subject = no
		is_in_faction = no
		is_faction_leader = no
		NOT = {
			TAG = PAN
			OR = {
				is_in_south_america = yes
				is_in_central_america = yes
			}
		}
		PAN = {
			OR = {
				is_in_south_america = yes
				is_in_central_america = yes
			}
			any_neighbor_country = {
				OR = {
					is_in_south_america = yes
					is_in_central_america = yes
				}
			}
		}
	}

	option = { #create the pact
		name = defensivepact.1.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				strength_ratio = {
					tag = PAN
					ratio < 1
				}
			}
			modifier = {
				factor = 0.5
				FROM = {
					any_allied_country = {
						is_major = yes
					}
				}
			}
		}
	}

	option = { #don't create the pact
		name = defensivepact.1.b
	}
}
