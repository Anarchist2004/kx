	id = gab.2
	immediate = { log = "[GetDateText]: [Root.GetName]: event gab.2" }
	title = gab.2.t
	desc = gab.2.d

	fire_only_once = yes

	is_triggered_only = yes

	option = {
		name = gab.2.a
		ai_chance = {
			factor = 50
		}
		create_country_leader = {
			name = "Gabriel Leon M'ba"
			picture = "Portrait_GAB_Gabriel_Leon_Mba.tga"
			expire = "1965.1.1"
			ideology = authoritarian_democrat_subtype
		}
		set_politics = {
			ruling_party = authoritarian_democrat
			elections_allowed = no
		}
		complete_national_focus = GAB_paranoia_of_man
	}

	option = {
		name = gab.2.b
		ai_chance = {
			factor = 50
		}
		create_country_leader = {
			name = "Charles N'Tchoréré"
			picture = "Portrait_GAB_Tchorere.tga"
			expire = "1965.1.1"
			ideology = authoritarian_democrat_subtype
		}
		set_party_name = {
			ideology = authoritarian_democrat
			long_name = GAB_authoritarian_democrat_party_2_long
			name = GAB_authoritarian_democrat_party_2
		}
		set_party_name = {
			ideology = paternal_autocrat
			long_name = GAB_paternal_autocrat_party_2_long
			name = GAB_paternal_autocrat_party_2
		}
		set_politics = {
			ruling_party = authoritarian_democrat
			elections_allowed = no
		}
		add_popularity = {
			ideology = social_conservative
			popularity = -0.2
		}
		complete_national_focus = GAB_sucsesful_coup
		if = {
			limit = { is_subject = yes }
			overlord = {
				set_autonomy = {
					target = GAB
					autonomy_state = autonomy_free
				}
			}
		}
		if = {
			limit = { is_in_faction = yes }
			leave_faction = yes
		}
		
		separate_peace_all_wars = yes
	}
}
