					NOT = {
						has_country_leader = {
							name = "Aleksey Rykov"
							ruling_only = yes
						}
					}
				}

					limit = {
						has_country_flag = soviet_russia_restored
						NOT = { has_government = totalist }
						NOT = {
							has_country_leader = {
								name = "Aleksey Rykov"
								ruling_only = yes
							}
						}
					}
					country_event = russoc.40
				}
				else_if = {
					limit = {
						NOT = { has_country_flag = soviet_russia_restored }
						NOT = { has_country_flag = RUS_KER }

					}

				}
			}
