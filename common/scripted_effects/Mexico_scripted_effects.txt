MEX_decrease_resistance = {
	if = {
		limit = { has_idea = MEX_anti_goldshirt_resistance_idea }
		swap_ideas = {
			remove_idea = MEX_anti_goldshirt_resistance_idea
			add_idea = MEX_anti_goldshirt_resistance_idea2
		}
	}
	else_if = {
		limit = { has_idea = MEX_anti_goldshirt_resistance_idea2 }
		swap_ideas = {
			remove_idea = MEX_anti_goldshirt_resistance_idea2
			add_idea = MEX_anti_goldshirt_resistance_idea3
		}
	}
	else_if = {
		limit = { has_idea = MEX_anti_goldshirt_resistance_idea3 }
		swap_ideas = {
			remove_idea = MEX_anti_goldshirt_resistance_idea3
			add_idea = MEX_anti_goldshirt_resistance_idea4
		}
	}
	else_if = {
		limit = { has_idea = MEX_anti_goldshirt_resistance_idea4 }
		remove_ideas = MEX_anti_goldshirt_resistance_idea4
	}
}
